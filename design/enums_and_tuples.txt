// Everyone loves enums! These are similar to Rust's enums.
enum PersonMood {
    Happy
    Sad
    Asleep
}

// You can have a value for types in an enum.
enum Things {
    Number( int )
    Text( string )
    Nothing
}

// To use an enum, you can do this:
let my_thing (Things) = Things.Number 20

// The value for these can be anything from structs to other enums. However,
// because of this syntax, you can only have one value per enum type. If you
// need more, use a struct or a tuple.

// Speaking of tuples, you can define them easily.
let my_tuple (uint string float) = (10 "hello, world!" 4.502)
